function getApiError(err, fallback) {
  const detail = err?.response?.data?.detail;
 
  if (typeof detail === "string") {
    return detail;
  }
 
  if (detail && typeof detail === "object") {
    return Object.values(detail).flat().join(" ");
  }
 
  return err?.message || fallback;
}
